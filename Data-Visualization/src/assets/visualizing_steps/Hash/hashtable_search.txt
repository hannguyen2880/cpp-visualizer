bool HashTable::searchItem(int key) {
    int index = hashFunction(key);
    int originalIndex = index;

    while (isFilled[index]) {
        if (table[index] == key) {
            return true;
        }
        index = (index + 1) % size;
        if (index == originalIndex) {
            return false;
        }
    }
    return false;
}
